WITH 
  MEMBER [Measures].[Percentage of transports delayed more than one day] AS 
    ([Measures].[Number of transport facts]) / 
    (
      [Measures].[Number of transport facts], 
      [ID Start Date].[Year].[All], 
      [Junk DT].[Delay Range].[All]
    ), 
    FORMAT_STRING = "Percent"
SELECT 
  NON EMPTY { 
    [Measures].[Percentage of transports delayed more than one day] 
  } ON COLUMNS 
FROM 
  ( 
    SELECT 
      ( { [Junk DT].[Delay Range].&[More] } ) ON COLUMNS 
    FROM 
      ( 
        SELECT 
          ( { [ID Start Date].[Year].&[2024] } ) ON COLUMNS 
        FROM 
          [Kolejnictwo Hurtownia] 
      ) 
  ) 
WHERE 
  ( [ID Start Date].[Year].&[2024], [Junk DT].[Delay Range].&[More] ) 
CELL PROPERTIES VALUE, BACK_COLOR, FORE_COLOR, FORMATTED_VALUE, FORMAT_STRING, FONT_NAME, FONT_SIZE, FONT_FLAGS

SELECT 
    NON EMPTY { [Measures].[Number of contracts] } ON COLUMNS
FROM 
    ( SELECT 
        { [Contract DT SCD2].[Expiration Year].&[2025] } ON COLUMNS 
        FROM ( SELECT 
            { [Contract DT SCD2].[Expiration Month].&[01] } ON COLUMNS 
            FROM ( SELECT 
                { [ID End Station].[Name].&[Aaronberg] } ON COLUMNS 
                FROM [Kolejnictwo Hurtownia]
            )
        )
    )
WHERE 
    ( 
        [ID End Station].[Name].&[Aaronberg], 
        [Contract DT SCD2].[Expiration Month].&[01], 
        [Contract DT SCD2].[Expiration Year].&[2025] 
    )
CELL PROPERTIES 
    VALUE, BACK_COLOR, FORE_COLOR, FORMATTED_VALUE, FORMAT_STRING, FONT_NAME, FONT_SIZE, FONT_FLAGS

7)
WITH 
  MEMBER [Measures].[Average Delays per Contract] AS 
    IIF(
      [Measures].[Number of contracts] = 0, 
      NULL, 
      ([Measures].[Number of transport facts]) / ([Measures].[Number of contracts])
    ), 
    FORMAT_STRING = "Standard"
SELECT 
  NON EMPTY { 
    [Measures].[Average Delays per Contract] 
  } ON COLUMNS 
FROM 
  ( 
    SELECT 
      ( 
        { [Junk DT].[Is Delay].&[True] } 
      ) ON COLUMNS 
    FROM 
      ( 
        SELECT 
          ( 
            { [ID Start Date].[Year].&[2023] } 
          ) ON COLUMNS 
        FROM 
          [Kolejnictwo Hurtownia] 
      ) 
  ) 
WHERE 
  ( 
    [ID Start Date].[Year].&[2023], 
    [Junk DT].[Is Delay].&[True] 
  ) 
CELL PROPERTIES VALUE, BACK_COLOR, FORE_COLOR, FORMATTED_VALUE, FORMAT_STRING, FONT_NAME, FONT_SIZE, FONT_FLAGS
